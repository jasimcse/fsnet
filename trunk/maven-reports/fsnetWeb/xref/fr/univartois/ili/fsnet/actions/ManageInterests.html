43      @Override
44      public ActionForward create(ActionMapping mapping, ActionForm form,
45              HttpServletRequest request, HttpServletResponse response)
46              throws IOException, ServletException {
47          EntityManager em = PersistenceProvider.createEntityManager();
48          DynaActionForm dynaForm = (DynaActionForm) form; // NOSONAR
49          InterestFacade facade = new InterestFacade(em);
50          String interestName = (String) dynaForm.get("createdInterestName");
51  
52          logger.info("new interest: " + interestName);
53  
54          try {
55              em.getTransaction().begin();
56              if (dynaForm.get("parentInterestId") != null
57                      && !((String) dynaForm.get("parentInterestId")).isEmpty()) {
58                  facade.createInterest(interestName, Integer.valueOf((String) dynaForm.get("parentInterestId")));
59              } else {
60                  facade.createInterest(interestName);
61              }
62  
63              em.getTransaction().commit();
64          } catch (RollbackException ex) {
65              ActionErrors actionErrors = new ActionErrors();
66              ActionMessage msg = new ActionMessage("interest.alreadyExists");
67              actionErrors.add("createdInterestName", msg);
68              saveErrors(request, actionErrors);
69          }
70  
71          em.close();
72  
73          dynaForm.set("createdInterestName", "");
74  
75          return mapping.findForward("success");
76      }

78      public ActionForward add(ActionMapping mapping, ActionForm form,
79              HttpServletRequest request, HttpServletResponse response)
80              throws IOException, ServletException {
81          EntityManager em = PersistenceProvider.createEntityManager();
82          DynaActionForm dynaForm = (DynaActionForm) form;// NOSONAR
83          int interestId = Integer.valueOf((String) dynaForm.get("addedInterestId"));
84  
85          SocialEntity user = UserUtils.getAuthenticatedUser(request, em);
86  
87          SocialEntityFacade facadeSE = new SocialEntityFacade(em);
88          InterestFacade facadeInterest = new InterestFacade(em);
89  
90          Interest interest = facadeInterest.getInterest(interestId);
91          if (interest != null) {
92              logger.info("add interest: id=" + interestId + " for user: "
93                      + user.getName() + " " + user.getFirstName() + " "
94                      + user.getId());
95  
96              em.getTransaction().begin();
97              facadeSE.addInterest(interest, user);
98              em.getTransaction().commit();
99          }
100         em.close();
101         ActionRedirect redirect = new ActionRedirect(mapping.findForward("success"));
102         redirect.addParameter("infoInterestId", interestId);
103         return redirect;
104     }

210             if (user.getInterests().contains(interest)) {
211                 request.setAttribute("own", true);
212             } else {
213                 request.setAttribute("own", false);
214             }
215 
216             for (Map.Entry<String, List<Interaction>> interactionEntry : resultMap.entrySet()) {
217                 request.setAttribute(interactionEntry.getKey(),

219             }
220         }
