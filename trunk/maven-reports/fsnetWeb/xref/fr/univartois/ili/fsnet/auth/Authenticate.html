22  /**
23   * This class represents a servlet that is used in order to authenticate members
24   * 
25   * @author Mathieu Boniface < mat.boniface {At} gmail.com >
26   */
27  public class Authenticate extends HttpServlet {
28  
29  	/**
30  	 * Welcome page path when the user is authenticated
31  	 */
32  	private static final String WELCOME_AUTHENTICATED_PAGE = "Home.do";
33  	/**
34  	 * Welcome page path when the user is NOT authenticated
35  	 */
36  	public static final String WELCOME_NON_AUTHENTICATED_PAGE = "/WEB-INF/jsp/login.jsp";
37  	/**
38  	 * Authenticated user key in session scope
39  	 */
40  	public static final String AUTHENTICATED_USER = "userId";
41  
42  	/**
43  	 * This method is called when an user user tries to sign in
44  	 */
45  	@Override
46  	protected void doGet(HttpServletRequest req, HttpServletResponse resp)
47  			throws ServletException, IOException {
48  		boolean authenticated = false;
49  		String memberMail = req.getParameter("memberMail");
50  		String memberPass = req.getParameter("memberPass");
51  		EntityManager em = PersistenceProvider.createEntityManager();;
52  		SocialEntity es = null;
53  		if (memberMail != null && memberPass != null) {
54  			SocialEntityFacade socialEntityFacade = new SocialEntityFacade(em);
55  			es = socialEntityFacade.findByEmail(memberMail);
56  			if (es != null
57  					&& Encryption.testPassword(memberPass, es.getPassword())) {
58  				authenticated = true;
59  				req.getSession(true).setAttribute(AUTHENTICATED_USER,
60  						es.getId());
61  				ManagePrivateMessages.refreshNumNewMessages(req, em);
62  				ManageContacts.refreshNumNewContacts(req, em);
63  			} else {
64  				req.setAttribute("loginMessage", "login.error");
65  			}
66  		}
67  
68  		if (authenticated) {
69  			// the user is now authenticated
70  			HttpSession session = req.getSession(true);
71  			String lastRequestedURL = (String) session.getAttribute("requestedURL");
72  			if (lastRequestedURL != null) {
73  				resp.sendRedirect(lastRequestedURL);
74  				session.removeAttribute("requestedURL");
75  			} else {
76  				resp.sendRedirect(WELCOME_AUTHENTICATED_PAGE);	
77  			}
78  			em.getTransaction().begin();
79  			SocialEntity user;
80  			user = em.find(SocialEntity.class, es.getId());
81  			user.setLastConnection(new Date());
82  			em.merge(user);
83  			em.getTransaction().commit();
84  			
85  			String userName = user.getFirstName() + " " + user.getName();
86  			LoggedUsersContainer.getInstance().addUser(user.getId(), userName);
87  		} else {
88  			// the user is not authenticated
89  			RequestDispatcher dispatcher = req
90  					.getRequestDispatcher(WELCOME_NON_AUTHENTICATED_PAGE);
91  			dispatcher.forward(req, resp);
92  		}
93  		em.close();
94  	}
95  
96  	/**
97  	 * Delegated to doGet
98  	 */
99  	@Override
100 	protected void doPost(HttpServletRequest req, HttpServletResponse resp)
101 			throws ServletException, IOException {
102 		doGet(req, resp);
103 	}
