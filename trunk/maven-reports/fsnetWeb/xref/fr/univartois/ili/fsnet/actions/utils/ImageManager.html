109 	private static void sendPicture(InputStream stream,
110 			HttpServletResponse response) throws IOException {
111 		byte[] datas = new byte[BUFFER_SIZE];
112 		int numRead = stream.read(datas);
113 		OutputStream out = response.getOutputStream();
114 		while (numRead != -1) {
115 			out.write(datas, 0 , numRead);
116 			numRead = stream.read(datas);
117 		}
118 		stream.close();
119 		out.flush();
120 	}

139 	private static BufferedImage getProperResizedImage(
140 			BufferedImage incommingImage, PictureType pictureType,
141 			int maxWidthOrHeight) throws FileNotFoundException, IOException {
142 		AffineTransform tx = new AffineTransform();
143 		double scaleValue;
144 		if (incommingImage.getWidth() > incommingImage.getHeight()) {
145 			scaleValue = maxWidthOrHeight / (double) incommingImage.getWidth();
146 		} else {
147 			scaleValue = maxWidthOrHeight / (double) incommingImage.getHeight();
148 		}
149 
150 		tx.scale(scaleValue, scaleValue);
151 		AffineTransformOp op = new AffineTransformOp(tx,
152 				AffineTransformOp.TYPE_BICUBIC);
153 		BufferedImage properResized = new BufferedImage((int) (incommingImage
154 				.getWidth() * scaleValue),
155 				(int) (incommingImage.getHeight() * scaleValue), pictureType
156 						.getImageType());
157 		op.filter(incommingImage, properResized);
158 		return properResized;
159 	}

161 	private static BufferedImage convert(BufferedImage image) {
162 		BufferedImage img = new BufferedImage(image.getWidth(), image
163 				.getHeight(), Transparency.TRANSLUCENT);
164 		Graphics2D g2 = (Graphics2D) img.getGraphics();
165 		g2.drawImage(image, 0, 0, null);
166 		g2.dispose();
167 		return img;
168 	}

179 	private static void installPicture(String fileName,
180 			PictureType pictureType, BufferedImage image)
181 			throws IllegalStateException {
182 		String directory = getStorageDirectory();
183 		if (directory != null) {
184 			removeOldPicture(directory + fileName);
185 			String fileToCreate = directory + fileName
186 					+ PictureType.PNG.getSuffix();
187 			File imageFile = new File(fileToCreate);
188 			try {
189 				OutputStream out = new FileOutputStream(imageFile);
190 				ImageIO.write(image, "png", out);
191 				out.flush();
192 				out.close();
193 			} catch (IOException e) {
194 				Logger.getAnonymousLogger().log(Level.SEVERE, "", e);
195 				throw new IllegalStateException(e);
196 			}
197 		} else {
198 			Logger.getAnonymousLogger().severe(
199 					"The storage directory for pictures is not configured");
200 			throw new IllegalStateException();
201 		}
202 	}
