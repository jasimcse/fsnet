45  	private Date validateDate(String eventDate, HttpServletRequest request,
46  			String propertyKey) {
47  		Date typedEventDate;
48  		try {
49  			typedEventDate = DateUtils.format(eventDate);
50  		} catch (ParseException e) {
51  			ActionErrors errors = new ActionErrors();
52  			errors.add(propertyKey, new ActionMessage(("event.date.errors")));
53  			saveErrors(request, errors);
54  			return null;
55  		}
56  		Calendar calendar = Calendar.getInstance();
57  		calendar.set(calendar.get(Calendar.YEAR), calendar.get(Calendar.MONTH),
58  				calendar.get(Calendar.DAY_OF_MONTH), 0, 0, -1);
59  		Date today = calendar.getTime();
60  		if (typedEventDate.before(today)) {
61  			ActionErrors errors = new ActionErrors();
62  			errors.add(propertyKey, new ActionMessage(("date.error.invalid")));
63  			saveErrors(request, errors);
64  			return null;
65  		}
66  		return typedEventDate;
67  	}

140 		EntityManager em = PersistenceProvider.createEntityManager();
141 		SocialEntity user = UserUtils.getAuthenticatedUser(request, em);
142 		InteractionFacade interactionFacade = new InteractionFacade(em);
143 		MeetingFacade meetingFacade = new MeetingFacade(em);
144 
145 		em.getTransaction().begin();
146 		Meeting meeting = meetingFacade.getMeeting(Integer.parseInt(eventId));
147 		Set<SocialEntity> followingEntitys = meeting.getFollowingEntitys();
148 		for (SocialEntity se : followingEntitys) {
149 			se.getFavoriteInteractions().remove(meeting);
150 		}
151 		InteractionRoleFacade interactionRoleFacade = new InteractionRoleFacade(
152 				em);
153 		interactionRoleFacade.unsubscribeAll(meeting);
154 		interactionFacade.deleteInteraction(user, meeting);
155 		try {
156 			em.getTransaction().commit();
157 		} catch (RollbackException e) {
158 			servlet.log("no commit", e);
159 		}
160 		em.close();
161 
162 		return mapping.findForward("success");
