49  		String interestName = (String) dynaForm.get("createdInterestName");

52  
53  		try {
54  			em.getTransaction().begin();
55  			if (dynaForm.get("parentInterestId") != null
56  					&& !((String) dynaForm.get("parentInterestId")).isEmpty()) {
57  				facade.createInterest(interestName, Integer
58  						.valueOf((String) dynaForm.get("parentInterestId")));
59  			} else {
60  				facade.createInterest(interestName);
61  			}
62  
63  			em.getTransaction().commit();
64  		} catch (RollbackException ex) {
65  			ActionErrors actionErrors = new ActionErrors();
66  			ActionMessage msg = new ActionMessage("interest.alreadyExists");
67  			actionErrors.add("createdInterestName", msg);
68  			saveErrors(request, actionErrors);
69  		}
70

72  		dynaForm.set("createdInterestName","");
