79  	public ActionForward saveMailConfiguration(ActionMapping mapping,
80  			ActionForm form, HttpServletRequest request,
81  			HttpServletResponse response) throws IOException, ServletException {
82  		DynaActionForm dynaform = (DynaActionForm) form; // NOSONAR
83  		EntityManager em = factory.createEntityManager();
84  		em.getTransaction().begin();
85  		if ("".equals(dynaform.get("enableTLS"))) {
86  			saveProperty(em, FSNetConfiguration.ENABLE_TLS_KEY, Boolean.FALSE
87  					.toString());
88  		} else {
89  			saveProperty(em, FSNetConfiguration.ENABLE_TLS_KEY, Boolean.TRUE
90  					.toString());
91  		}
92  		if ("".equals(dynaform.get("enableSSL"))) {
93  			saveProperty(em, FSNetConfiguration.SSL_KEY, Boolean.FALSE
94  					.toString());
95  		} else {
96  			saveProperty(em, FSNetConfiguration.SSL_KEY, Boolean.TRUE
97  					.toString());
98  		}
99  		if ("".equals(dynaform.get("enableAuthentication"))) {
100 			saveProperty(em, FSNetConfiguration.ENABLE_AUTHENTICATION_KEY,
101 					Boolean.FALSE.toString());
102 		} else {
103 			saveProperty(em, FSNetConfiguration.ENABLE_AUTHENTICATION_KEY,
104 					Boolean.TRUE.toString());
105 			saveProperty(em, FSNetConfiguration.SMTP_USER_KEY,
106 					(String) dynaform.get("SMTPUsername"));
107 			saveProperty(em, FSNetConfiguration.SMTP_PASSWORD_KEY,
108 					(String) dynaform.get("SMTPPassword"));
109 		}
110 		saveProperty(em, FSNetConfiguration.MAIL_FROM_KEY, (String) dynaform
111 				.get("MailFrom"));
112 		saveProperty(em, FSNetConfiguration.SMTP_HOST_KEY, (String) dynaform
113 				.get("SMTPHost"));
114 		String SMTP_Port = (String) dynaform.get("SMTPPort");
115 		try {
116 			Integer.parseInt(SMTP_Port);
117 		} catch (NumberFormatException e) {
118 			SMTP_Port = DEFAULT_SMTP_PORT;
119 		}
120 		saveProperty(em, FSNetConfiguration.SMTP_PORT_KEY, SMTP_Port);
121 		saveProperty(em, FSNetConfiguration.FSNET_WEB_ADDRESS_KEY,
122 				(String) dynaform.get("FSNetWebURL"));
123 		String dirName = (String) dynaform.get("PicturesDirectory");
124 		if (isValidDirectory(dirName)) {
125 			saveProperty(em, FSNetConfiguration.PICTURES_DIRECTORY_KEY, dirName);
126 		} else {
127 			ActionErrors errors = new ActionErrors();
128 			errors.add("PicturesDirectory", new ActionMessage("configure.23"));
129 			saveErrors(request, errors);
130 		}
131 		em.getTransaction().commit();
132 		em.close();
133 		FSNetConfiguration.getInstance().refreshConfiguration();
134 		return mapping.findForward("success");
135 	}

137 	private void saveProperty(EntityManager em, String key, String value) {
138 		Property property = em.find(Property.class, key);
139 		if (property == null) {
140 			property = new Property();
141 			property.setKey(key);
142 			property.setValue(value);
143 			em.persist(property);
144 		} else {
145 			property.setValue(value);
146 			em.merge(property);
147 		}
148 	}
