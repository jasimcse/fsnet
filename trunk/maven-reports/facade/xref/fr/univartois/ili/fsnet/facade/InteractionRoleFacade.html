28       * Subscribe to an interaction
29       * @param se
30       * @param i
31       */
32      public final void subscribe(SocialEntity se, Interaction i) {
33          if (se == null || i == null) {
34              throw new IllegalArgumentException();
35          }
36          InteractionRole interactionRole = new InteractionRole(se, i);
37          interactionRole.setRole(InteractionRole.RoleName.SUBSCRIBER);
38          se.getRolesInInteractions().add(interactionRole);
39          i.getRoles().add(interactionRole);
40          em.persist(interactionRole);
41      }

43      /**
44       * Unsubscribe from an interaction
45       * @param se
46       * @param i
47       */
48      public final void unsubscribe(SocialEntity se, Interaction i) {
49          if (se == null || i == null) {
50              throw new IllegalArgumentException();
51          }
52          InteractionRolePK interactionRolePK = new InteractionRolePK();
53          interactionRolePK.setInteractionId(i.getId());
54          interactionRolePK.setSocialEntityId(se.getId());
55          InteractionRole interactionRole = em.find(InteractionRole.class, interactionRolePK);
56          se.getRolesInInteractions().remove(interactionRole);
57          i.getRoles().remove(interactionRole);
58          em.remove(interactionRole);
59      }

62       *

64       */
65      public final void unsubscribeAll(Interaction i){
66          if(i == null) throw new IllegalArgumentException();
67          List<SocialEntity> socialEntities = new ArrayList<SocialEntity>();
68          for(InteractionRole interactionRole : i.getRoles()){
69              socialEntities.add(interactionRole.getSocialEntity());
70          }
71          for(SocialEntity se : socialEntities){
72              unsubscribe(se, i);
73          }

83          if (se == null || i == null) {

86          for (InteractionRole interactionRole : se.getRolesInInteractions()) {
87              if (interactionRole.getInteraction().equals(i)

89                  return true;
