17              hash = MessageDigest.getInstance("MD5").digest(uniqueKey);
18          } catch (NoSuchAlgorithmException e) {
19              throw new InternalError("no MD5 support in this VM");
20          }
21          StringBuilder hashString;
22          hashString = new StringBuilder();
23          for (int i = 0; i < hash.length; ++i) {
24              String hex;
25              hex = Integer.toHexString(hash[i]);
26              if (hex.length() == 1) {
27                  hashString.append('0');
28                  hashString.append(hex.charAt(hex.length() - 1));
29              } else {
30                  hashString.append(hex.substring(hex.length() - 2));
31              }
32          }
33          return hashString.toString();
34      }
