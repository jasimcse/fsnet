22  /**
23   * The class TrayLauncher creates an instance of the web service and an instance
24   * of FSNetTray and add the tray to the system tray
25   * 
26   * @author Matthieu Proucelle <matthieu.proucelle at gmail.com>
27   */
28  public class TrayLauncher {
29  
30      private static ResourceBundle trayi18n;
31      private static Logger logger = Logger.getLogger(TrayLauncher.class.getName());
32      private static FSNetTray tray;
33      private static WSConnector connector;
34      private static WSControl control;
35      private static ConfigurationFrame confFrame;
36  
37      /**
38       * Open a configuration frame
39       */
40      public static void showConfigFrame() {
41          confFrame = new ConfigurationFrame(control);
42          connector.addListener(confFrame);
43          confFrame.show();
44      }
45  
46      /**
47       * Called when the configuration is done
48       */
49      public static final void configurationValidated() {
50          connector.removeListener(confFrame);
51          confFrame = null;
52          reload();
53      }
54  
55      private TrayLauncher() {
56      }
57  
58      public static void main(String[] args) throws IOException {
59          if (!SystemTray.isSupported()) {
60              JOptionPane.showMessageDialog(null, "SystemTray is not supported");
61          } else {
62              try {
63                  UIManager.setLookAndFeel(UIManager.getSystemLookAndFeelClassName());
64              } catch (Exception e) {
65                  logger.log(Level.INFO, "Unable to change L&F");
66              }
67              Options.loadOptions();
68              connector = new WSConnector();
69              control = new WSControl(connector);
70              reload();
71          }
72      }
73  
74      /**
75       * Load or reaload the tray
76       */
77      public static final void reload() {
78          if (tray != null) {
79              connector.removeListener(tray);            
80              SystemTray.getSystemTray().remove(tray.getTrayIcon());
81          }
82  
83          SwingUtilities.invokeLater(new Runnable() {
84  
85              @Override
86              public void run() {
87  
88  
89                  trayi18n = ResourceBundle.getBundle("ressources/Trayi18n", Options.getLocale());
90                  if (!Options.isConfigured()) {
91                      showConfigFrame();
92                  } else {
93                      ImageIcon icon = getImageIcon("/ressources/iconefsnet.png");
94                      if (icon != null) {
95                          try {
96                              tray = new FSNetTray(icon.getImage(), control);
97                              connector.addListener(tray);
98                              SystemTray.getSystemTray().add(tray.getTrayIcon());
99                          } catch (AWTException ex) {
100                             logger.log(Level.SEVERE, null, ex);
101                         }
102                     }
103                 }
104             }
105         });
106     }
107 
108     /**
109      *
110      * @return the i18n bundle
111      */
112     public static final ResourceBundle getBundle() {
113         return trayi18n;
114     }
115 
116     /**
117      *
118      * @param path
119      * @return the image to display in the tray
120      */
121     public static ImageIcon getImageIcon(String path) {
122         URL imageURL = FSNetTray.class.getResource(path);
123         if (imageURL == null) {
124             logger.log(Level.SEVERE, "Resource not found: " + path);
125             return null;
126         } else {
127             return new ImageIcon(imageURL);
128         }
