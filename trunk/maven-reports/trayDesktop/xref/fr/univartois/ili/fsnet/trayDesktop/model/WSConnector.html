108 	/**
109 	 * Check if the config is valid
110 	 * 
111 	 * @return
112 	 */
113 	private boolean verifConfig() {
114 		if (!Options.isConfigured()) {
115 			return false;
116 		}
117 
118 		ExecutorService executor = Executors.newSingleThreadExecutor();
119 		Future<Boolean> future = executor.submit(new Callable<Boolean>() {
120 
121 			@Override
122 			public Boolean call() {
123 				try {
124 					InfoService infoService = new InfoService(new URL(Options
125 							.getWSUrl()), new QName(
126 							"http://webservice.fsnet.ili.univartois.fr/",
127 							"InfoService"));
128 					infoPort = infoService.getInfoPort();
129 					return infoPort.isMember(Options.getLogin(), Options
130 							.getPassword());
131 				} catch (MalformedURLException ex) {
132 					Logger.getLogger(WSConnector.class.getName()).log(
133 							Level.SEVERE, null, ex);
134 				}
135 				return false;
136 			}
137 		});
138 		try {
139 			return future.get(5, TimeUnit.SECONDS);
140 		} catch (Exception e) {
141 		}
142 		return false;
143 	}

146 	 * Try to change the config and notify listeners with the result

149 	 * @param fsnetUrl

155 	public void changeConfig(final String wSUrl, final String fsnetUrl,
156 			final String login, final String password, final LANG lang,
157 			final int lag) {
158 		try {
159 			Thread r = new Thread() {
160 
161 				@Override
162 				public void run() {
163 					try {
164 						InfoService infoService = new InfoService(
165 								new URL(wSUrl),
166 								new QName(
167 										"http://webservice.fsnet.ili.univartois.fr/",
168 										"InfoService"));
169 						infoPort = infoService.getInfoPort();
170 						if (infoPort.isMember(login, password)) {
171 							Options.setWSUrl(wSUrl);
172 							Options.setFsnetUrl(fsnetUrl);
173 							Options.setLogin(login);
174 							Options.setPassword(password);
175 							Options.setLanguage(lang);
176 							Options.setLag(lag);
177 							Options.saveOptions();
178 							startNotifications();
179 							fireConnection("VALIDCONFIGURATION");
180 						} else {
181 							fireError("BADLOGIN");
182 						}
183 					} catch (Exception ex) {
184 						logger.log(Level.SEVERE, null, ex);
185 						fireError("NOCONNECTION");
186 					}
187 				}
188 			};
189 			r.start();
190 			r.join(5000);
191 		} catch (InterruptedException ex) {
192 			Logger.getLogger(WSConnector.class.getName()).log(Level.SEVERE,
193 					null, ex);
194 			fireError("NOCONNECTION");
195 		}
196 	}
