22  public class IsAuthenticatedFilter implements Filter {
23

26  	/**
27  	 * Verify if the current user is authenticated
28  	 */
29  	public void doFilter(final ServletRequest req,
30  			final ServletResponse response, final FilterChain chain)
31  			throws IOException, ServletException {
32  		HttpSession session;
33  
34  		RequestDispatcher dispatch;
35  		HttpServletRequest request = (HttpServletRequest) req;
36  		session = request.getSession(); // NOSONAR
37  		Integer userId = (Integer) session
38  				.getAttribute(Authenticate.AUTHENTICATED_USER);
39  
40  		if (userId == null) {
41  			// Reconstruct the requested url and store it in the session
42  			StringBuffer requestedURL = ((HttpServletRequest) request)
43  					.getRequestURL();
44  			if (request.getQueryString() != null) {
45  				requestedURL.append('?');
46  				requestedURL.append(request.getQueryString());
47  			}
48  			session.setAttribute("requestedURL", requestedURL.toString());
49  
50  			dispatch = servletContext
51  					.getRequestDispatcher(Authenticate.WELCOME_NON_AUTHENTICATED_PAGE);
52  			dispatch.forward(request, response);
53  		} else {
54  			chain.doFilter(request, response);
55  		}
56  	}
