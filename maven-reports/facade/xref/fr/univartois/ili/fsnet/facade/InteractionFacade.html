103 	public final List<Triple> getLastInteractions(SocialEntity user){
104 
105 		TypedQuery<Interaction> query = em.createQuery(
106 				"SELECT inter FROM Interaction inter, SocialEntity se, IN(se.contacts) c " +
107 				"WHERE inter.creator= c AND se = :user ORDER BY inter.lastModified DESC",Interaction.class);
108 		query.setParameter("user", user);
109 		query.setMaxResults(10);
110 		
111 		List<Interaction> result = query.getResultList();
112 		List<Triple> triples = new ArrayList<Triple>();
113 		
114 		String path = null;
115 		String id = null;
116 		for (Interaction interaction : result) {
117 			String clazz = interaction.getClass().getSimpleName(); 
118 			if ("Announcement".equals(clazz)) {
119 				path = "/DisplayAnnounce";
120 				id ="idAnnounce";
121 			} else if ("Meeting".equals(clazz)) {
122 				path = "/DisplayEvent";
123 				id = "eventId";
124 			} else if ("Topic".equals(clazz)) {
125 				path = "/Topic";
126 				id = "topicId";
127 			} else if ("Hub".equals(clazz)) {
128 				path = "/DisplayHub";
129 				id = "hubId";
130 			} else if ("Community".equals(clazz)) {
131 				path = "/DisplayCommunity";
132 				id = "communityId";
133 			}
134 			triples.add(new Triple(interaction,path,id));
135 		}
136 
137 		return triples;
138 	}
