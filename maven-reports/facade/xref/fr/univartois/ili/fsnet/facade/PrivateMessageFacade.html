26       * @param from the author of the message
27       * @param subject the subject of the message
28       * @param to the recipient of the message
29       * @return the persisted PrivateMessage

31      public final PrivateMessage sendPrivateMessage(String body, SocialEntity from, String subject, SocialEntity to) {
32          PrivateMessage pm = new PrivateMessage(body, from, subject, to);
33          em.persist(pm);
34          from.getSentPrivateMessages().add(pm);
35          to.getReceivedPrivateMessages().add(pm);
36          return pm;
37      }

39      /**
40       * Delete a PrivateMessage
41       * @param entity the related entity (author or recipient)
42       * @param message the message to delete
43       */
44      public final void deletePrivateMessage(SocialEntity entity, PrivateMessage message) {
45          if (entity == null || message == null) {
46              throw new IllegalArgumentException();
47          }
48          if (message.getFrom().equals(entity)) {
49              entity.getSentPrivateMessages().remove(message);
50          }
51          if (message.getTo().equals(entity)) {
52              entity.getReceivedPrivateMessages().remove(message);
53          }
54      }

65          }
66          TypedQuery<PrivateMessage> query = em.createQuery(
67                  "SELECT message FROM PrivateMessage message WHERE message.to = :entity "
68                  + "AND (message.body LIKE :pattern OR message.subject LIKE :pattern", PrivateMessage.class);
