34  public class FSNetTray implements WSListener {

36      private final ResourceBundle trayi18n = TrayLauncher.getBundle();
37      private TrayIcon tray;

40      /**
41       *
42       * @param image the icon to display in system tray
43       * @param infoService the webservice to invoke
44       */
45      public FSNetTray(Image image, WSControl control) {
46          if (image == null || control == null) {
47              throw new IllegalArgumentException();
48          }
49          tray = new TrayIcon(image);
50          initTrayIcon();
51          this.control = control;
52      }

55       *
56       * @return the builded tray icon
57       */
58      public TrayIcon getTrayIcon() {
59          return tray;
60      }

62      /**
63       * Initialize tray icon's parameters
64       */
65      private void initTrayIcon() {
66          tray.setToolTip(trayi18n.getString("NOTIFICATIONFSNET"));
67          // Create a popup menu components
68          final PopupMenu popup = new PopupMenu();
69  
70          MenuItem configItem = new MenuItem(trayi18n.getString("CONFIGURATION"));
71          configItem.addActionListener(new ActionListener() {
72  
73              @Override
74              public void actionPerformed(ActionEvent e) {
75                  TrayLauncher.showConfigFrame();
76              }
77          });
78  
79          // TODO mettre ca dans la config
80          CheckboxMenuItem size = new CheckboxMenuItem(trayi18n.getString("SETAUTOSIZE"));
81          size.addItemListener(new ItemListener() {
82  
83              @Override
84              public void itemStateChanged(ItemEvent e) {
85                  int cb1Id = e.getStateChange();
86                  if (cb1Id == ItemEvent.SELECTED) {
87                      tray.setImageAutoSize(true);
88                  } else {
89                      tray.setImageAutoSize(false);
90                  }
91              }
92          });
93  
94          MenuItem exitItem = new MenuItem(trayi18n.getString("EXIT"));
95          exitItem.addActionListener(new ActionListener() {
96  
97              @Override
98              public void actionPerformed(ActionEvent e) {
99                  System.exit(0);
100             }
101         });
102         tray.addActionListener(new ActionListener() {
103 
104             @Override
105             public void actionPerformed(ActionEvent e) {
106                 try {
107                     Desktop.getDesktop().browse(new URI(Options.getFsnetUrl()));
108                 } catch (URISyntaxException ex) {
109                     Logger.getLogger(FSNetTray.class.getName()).log(Level.SEVERE, null, ex);
110                 } catch (IOException ex) {
111                     Logger.getLogger(FSNetTray.class.getName()).log(Level.SEVERE, null, ex);
112                 }
113             }
114         });
115 
116         tray.addMouseListener(new MouseAdapter() {
117 
118             @Override
119             public void mouseClicked(MouseEvent e) {
120                 //control.checkWS();
121                 Point position = e.getLocationOnScreen();
122                 control.check(position);
123             }
124         });
125         // Add components to popup menu
126         popup.add(configItem);
127         popup.add(size);
128         popup.add(exitItem);
129         tray.setPopupMenu(popup);
130     }
131
