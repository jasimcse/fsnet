40  	@Override
41  	public ActionForward create(ActionMapping mapping, ActionForm form,
42  			HttpServletRequest request, HttpServletResponse response)
43  			throws IOException, ServletException {
44  		DynaActionForm dynaForm = (DynaActionForm) form; // NOSONAR
45  		String to = dynaForm.getString("messageTo");
46  		String subject = dynaForm.getString("messageSubject");
47  		String body = dynaForm.getString("messageBody");
48  		EntityManager em = PersistenceProvider.createEntityManager();
49  		em.getTransaction().begin();
50  		SocialEntity authenticatedUser = UserUtils.getAuthenticatedUser(
51  				request, em);
52  		SocialEntityFacade sef = new SocialEntityFacade(em);
53  		StringTokenizer stk = new StringTokenizer(to,",");
54  		List<SocialEntity> receivers = new ArrayList<SocialEntity>();
55  		while(stk.hasMoreTokens()){
56  			String email = stk.nextToken();
57  			SocialEntity findByEmail = sef.findByEmail(email);
58  			if (findByEmail == null) {
59  				ActionErrors errors = new ActionErrors();
60  				errors.add("messageTo", new ActionMessage("privatemessages.to.error",email));
61  				saveErrors(request, errors);
62  				em.getTransaction().commit();
63  				em.close();
64  				return mapping.getInputForward();
65  			}
66  			receivers.add(findByEmail);
67  		}
68  		for (SocialEntity se : receivers  ) {
69  			PrivateMessageFacade pmf = new PrivateMessageFacade(em);
70  			pmf.sendPrivateMessage(body, authenticatedUser, subject,se);
71  		} 			
72  		em.getTransaction().commit();
73  		em.close();
74  		return mapping.findForward("success");
75  	}

227 	private static final int calculateNumNewMessage(
228 			Collection<PrivateMessage> messages) {
229 		int numNonReedPrivateMessages = 0;
230 		for (PrivateMessage pm : messages) {
231 			if (!pm.isReed()) {
232 				numNonReedPrivateMessages++;
233 			}
234 		}
235 		return numNonReedPrivateMessages;
236 	}

239 			Collection<PrivateMessage> messages) {
240 		HttpSession session = request.getSession();
241 		int numNonReedPrivateMessages = calculateNumNewMessage(messages);
242 		session.setAttribute("numNonReedPrivateMessages",
