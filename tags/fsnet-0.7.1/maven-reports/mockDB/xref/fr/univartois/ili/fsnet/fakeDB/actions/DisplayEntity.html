49  		StringBuilder url = new StringBuilder("http://yuml.me/diagram/class/");
50  		IdentifiableType<?> superEntity = entity.getSupertype();
51  		IdentifiableType<?> subEntity = entity;
52  		url.append(getEntityDescription(metamodel, entity.getJavaType(), "orange"));

64  		}

66  				.getDeclaredPluralAttributes()) {
67  			url.append(getEntityDescription(metamodel, entity.getJavaType(), "orange"));
68  			url.append(buildRelation(pluralAttribute.getPersistentAttributeType()));

70  			url.append(getEntityDescription(metamodel, pluralAttribute
71  					.getElementType().getJavaType(), null));
72  			url.append(", ");
73  		}
74  		for (SingularAttribute<?, ?> singularAttribute : entity.getDeclaredSingularAttributes()) {
75  			if (singularAttribute.getType().getPersistenceType().equals(PersistenceType.ENTITY)) {
76  				url.append(getEntityDescription(metamodel, entity.getJavaType(), "orange"));
77  				url.append(buildRelation(singularAttribute.getPersistentAttributeType()));
78  				url.append(getEntityDescription(metamodel, singularAttribute.getType().getJavaType(), null));

85  	public static String buildRelation(PersistentAttributeType attributeType) {
86  		if (PersistentAttributeType.MANY_TO_MANY.equals(attributeType)) {
87  			return "<*-*>";
88  		} else if (PersistentAttributeType.ONE_TO_MANY.equals(attributeType)) {
89  			return "<1-*>";
90  		} else if (PersistentAttributeType.MANY_TO_ONE.equals(attributeType)) {
91  			return "<*-1>";
92  		} else if (PersistentAttributeType.ONE_TO_ONE.equals(attributeType)) {
93  			return "<1-1>";
94  		} else if (PersistentAttributeType.EMBEDDED.equals(attributeType)) {
95  			return "<1-1>";
96  		}
97  		return null;
