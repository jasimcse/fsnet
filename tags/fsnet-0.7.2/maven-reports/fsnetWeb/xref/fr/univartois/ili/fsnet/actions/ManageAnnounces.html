45      @Override
46      public ActionForward create(ActionMapping mapping, ActionForm form,
47              HttpServletRequest request, HttpServletResponse response)
48              throws IOException, ServletException {
49          EntityManager entityManager = PersistenceProvider.createEntityManager();
50          SocialEntity user = UserUtils.getAuthenticatedUser(request,
51                  entityManager);
52  
53          DynaActionForm formAnnounce = (DynaActionForm) form; // NOSONAR
54          String title = (String) formAnnounce.get("announceTitle");
55          String content = (String) formAnnounce.get("announceContent");
56          String stringExpiryDate = (String) formAnnounce.get("announceExpiryDate");
57          String InterestsIds[] = (String[]) formAnnounce.get("selectedInterests");
58  
59          Announcement createdAnnounce;
60  
61          try {
62              Date expiryDate = DateUtils.format(stringExpiryDate);
63              if (0 > DateUtils.compareToToday(expiryDate)) {
64                  AnnouncementFacade announcementFacade = new AnnouncementFacade(entityManager);
65                  entityManager.getTransaction().begin();
66                  createdAnnounce = announcementFacade.createAnnouncement(user, title, content, expiryDate, false);
67                  InterestFacade fac = new InterestFacade(entityManager);
68                  List<Interest> interests = new ArrayList<Interest>();
69                  int currentId;
70                  for (currentId = 0; currentId < InterestsIds.length; currentId++) {
71                      interests.add(fac.getInterest(Integer.valueOf(InterestsIds[currentId])));
72                  }
73                  InteractionFacade ifacade = new InteractionFacade(entityManager);
74                  ifacade.addInterests(createdAnnounce, interests);
75                  entityManager.getTransaction().commit();
76              } else {
77                  ActionMessages errors = new ActionErrors();
78                  errors.add("message", new ActionMessage("error.dateBelowDateToday"));
79                  saveErrors(request, errors);
80                  entityManager.close();
81                  return mapping.findForward("failer");
82              }
83          } catch (ParseException e) {
84              servlet.log(getClass().getName()
85                      + " methode:create exception when formating date ");
86              return mapping.findForward("failer");
87          }
88          entityManager.close();
89          return mapping.findForward("success");
90      }
